40 bool hal::uc::msp430f2132::I2c_slave::init(uint8_t module, uint8_t ownAddress)
41 {
42  // SELECT MODULE -> This MCU has only 1 I2C module
43  if(module != 0)
44  {
45  setErrorCode(error_moduleDoesNotExist);
46  return false;
47  }
48 
49 
50  UCB0CTL1 |= UCSWRST; // Disable I2C module while configuring
51 
52 
53  // GPIO CONFIGURATION
54  P3SEL |= 0x06; // Use P3.1 and P3.2 for I2C
55 
56  // CONFIGURATION
57  UCB0CTL0 = UCMODE_3 + UCSYNC; // Master, I2C, Syncronous
58 
59  // CLOCK PRESCALE
60  UCB0BR0 = 80;
61  UCB0BR1 = 0;
62 
63  // OWN ADDRESS
64  UCB0I2COA = ownAddress; // Own adress. General call response disabled
65 
66 
67  UCB0CTL1 &= ~UCSWRST; // Clear SW reset, resume operation
68 
69 
70  // IRQ
71  UCB0I2CIE |= UCSTTIE; // Enable Start interrupt
72  IFG2&= ~(UCB0TXIFG + UCB0RXIFG); // Clear IRQ flags
73  IE2 |= UCB0RXIE + UCB0TXIE; // Enable RX IRQ and TX IRQ
74 
75 
76  setErrorCode(noError);
77  setInitialized();
78  return true;
79 }
