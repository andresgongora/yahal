195 void yahal::mcu::hwemulation::I2CMaster::emulateStart(void)
196 {
197  bool readBit = direction_ == Direction::READ ? true : false;
198  uint8_t composedAddress = (slave_address_ << 1) + readBit;
199 
200 
201  sendStart(); // Send start sequence
202  sendByte(composedAddress); // Send address
203  bool nack = receiveAck(); // Get: NACK = 1; ACK = 0;
204 
205 
206  if(nack)
207  {
208  received_nack_ = true; // NACK IRQ
209  direction_ = Direction::WRITE; // IF READING, ABORT IT: This is needed so that in the emulaiton loop a STOP is sent as soon as possible
210  }
211  else if(direction_ == Direction::WRITE )
212  {
213  buffer_TX_status_ = BufferStatus::JUST_READ; // TX IRQ
214  }
215 }

295 void yahal::mcu::hwemulation::I2CMaster::emulateLoop(void)
296 {
297  // While no STOP is sent, keep emulating HW features
298  bool busy = true;
299 
300  while(busy)
301  {
302  // NACK IRQ
303  if(received_nack_)
304  {
305  received_nack_ = false;
306  emulateIRQNack();
307  }
308 
309  // RX IRQ
310  else if(buffer_RX_status_ == BufferStatus::JUST_WRITTEN)
311  {
312  buffer_RX_status_ = BufferStatus::FULL;
313  emulateIRQRX();
314  }
315 
316  // TX IRQ
317  else if(buffer_TX_status_ == BufferStatus::JUST_READ)
318  {
319  buffer_TX_status_ = BufferStatus::EMPTY;
320  emulateIRQTX();
321  }
322 
323  // STOP
324  else if(send_stop_ && direction_ == Direction::WRITE)
325  {
326  send_stop_ = false;
327  emulateStop();
328  busy = false;
329  }
330 
331  // START
332  else if(end_start_)
333  {
334  busy = true;
335  end_start_ = false;
336  emulateStart();
337  }
338 
339  // READ
340  else if(direction_ == Direction::READ)
341  {
342  emulateReceiveBufferRX();
343  }
344 
345  // WRITE
346  else if(buffer_TX_status_ == BufferStatus::JUST_WRITTEN)
347  {
348  emulateSendBufferTX();
349  buffer_TX_status_ = BufferStatus::JUST_READ;
350  }
351 
352  // DEFAULT: STOP
353  else
354  {
355  send_stop_ = false;
356  emulateStop();
357  busy = false;
358  }
359  }
360 
361  // CLEAR ALL JUST TO BE ON THE SAFE SIDE
362  emulateReset();
363 
364 }
