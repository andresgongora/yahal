46 void yahal::mcu::targets::msp430f5309::I2C_multimaster::doInit(void)
47 {
48  UCB1CTL1 |= UCSWRST;
49  P4SEL |= BIT1 + BIT2;
50 
51  UCB1CTL0 = UCMST + UCMODE_3 + UCSYNC + UCMM;
52  UCB1CTL1 = UCSSEL__SMCLK + UCSWRST + UCTR;
