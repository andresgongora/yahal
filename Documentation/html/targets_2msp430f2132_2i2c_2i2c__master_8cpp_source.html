44 bool hal::uc::msp430f2132::I2c_master::init(uint8_t ownAddress)
45 {
46  UCB0CTL1 |= UCSWRST; // Disable I2C module while configuring
47 
48 
49  // GPIO CONFIGURATION
50  P3SEL |= 0x06; // Use P3.1 and P3.2 for I2C
51 
52  // CONFIGURATION
53  UCB0CTL0 = UCMST + UCMODE_3 + UCSYNC; // Master, I2C, Syncronous
54  UCB0CTL1 = UCSSEL_2 + UCTR + UCSWRST; // Use SMCLK as clock. Start in TX mode. Hold in SW reset
55 
56  // CLOCK PRESCALE
57  UCB0BR0 = 20;
58  UCB0BR1 = 0;
59 
60  // OWN ADDRESS
61  UCB0I2COA = ownAddress; // Own adress. General call response disabled
62  UCB0I2CSA = 0; // Slave address to write to. Set to 0 for now.
63 
64 
65  UCB0CTL1 &= ~UCSWRST; // Clear SW reset, resume operation
66 
67 
68  // IRQ
69  UCB0I2CIE |= UCNACKIE; // IRQ for: nack
70  IFG2&= ~(UCB0TXIFG + UCB0RXIFG); // Clear IRQ flags
71  IE2 |= UCB0RXIE + UCB0TXIE; // Enable RX IRQ and TX IRQ
72  msp430f2132::irq.i2c.subscribe(*this); // Subscribe to IRQ notifications
73 
74 
75  setErrorCode(Error::NO_ERROR);
76  setInitialized();
77  return true;
78 }
