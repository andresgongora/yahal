56 void SetVCoreUp (unsigned int level)
57 {
58  // Open PMM registers for write access
59  PMMCTL0_H = 0xA5;
60  // Make sure no flags are set for iterative sequences
61 // while ((PMMIFG & SVSMHDLYIFG) == 0);
62 // while ((PMMIFG & SVSMLDLYIFG) == 0);
63  // Set SVS/SVM high side new level
64  SVSMHCTL = SVSHE + SVSHRVL0 * level + SVMHE + SVSMHRRL0 * level;
65  // Set SVM low side to new level
66  SVSMLCTL = SVSLE + SVMLE + SVSMLRRL0 * level;
67  // Wait till SVM is settled
68  while ((PMMIFG & SVSMLDLYIFG) == 0);
69  // Clear already set flags
70  PMMIFG &= ~(SVMLVLRIFG + SVMLIFG);
71  // Set VCore to new level
72  PMMCTL0_L = PMMCOREV0 * level;
73  // Wait till new level reached
74  if ((PMMIFG & SVMLIFG))
75  while ((PMMIFG & SVMLVLRIFG) == 0);
76  // Set SVS/SVM low side to new level
77  SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level;
78  // Lock PMM registers for write access
79  PMMCTL0_H = 0x00;
80 }
