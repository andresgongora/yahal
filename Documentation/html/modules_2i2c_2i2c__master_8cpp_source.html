119 void yahal::mcu::modules::I2CMaster::handleBufferTXEmpty(void)
120 {
121  // This fuction might be entered because
122  // - Need to send register address, regardless if it is Write or Read -> Send RegAddr
123  // - This is a Read and the register address has just been sent -> Restart
124  // - This is a Write and there is data to be sent -> Send data
125  // - All data has been sent (last byte just went out) -> Stop
126 
127  if(send_register_address_pending_)
128  {
129  send_register_address_pending_ = false;
130  writeBufferTX(register_address_);
131  }
132  else if(direction_ == Direction::READ)
133  {
134  sendStart(); //restart
135  }
136  else if(pendingTransmissions())
137  {
138  // STORE DATA IN CASE ANOTHER IRQ HAPPENS
139  uint8_t currentTransmission = num_transmitted_;
140  num_transmitted_++;
141 
142  // SEND DATA
143  writeBufferTX(p_buffer_[currentTransmission]);
144  }
145  else // Nothing left to be done
146  {
147  stop();
148  }
149 }

153 void yahal::mcu::modules::I2CMaster::handleBufferRXFull(void)
154 {
155  // STORE DATA IN CASE ANOTHER IRQ HAPPENS
156  uint8_t currentTransmission = num_transmitted_;
157  num_transmitted_++;
158 
159  // PREPRARE TO RECEIVE LAST BYTE
160  if(pendingTransmissions() == 1) // Last byte is waiting in the shift register
161  {
162  stop(); // Send stop as soon as the last byte enters the RX buffer
163  }
164 
165  volatile uint8_t receivedByte = readBufferRX(); // Free RX buffer
166 
167 
168  if(currentTransmission >= num_transmissions_)
169  {
170  setErrorCode(Error::READ_OVERFLOW_ATTEMPT);
171  }
172  else
173  {
174  p_buffer_[currentTransmission] = receivedByte;
175  }
176 }

195 bool yahal::mcu::modules::I2CMaster::transmit(void)
196 {
197  // CHECK FOR ERRORS
198  if (slave_address_ & 0x80) {setErrorCode(Error::SLAVE_ADDRESS_NOT_7_BIT);}
199  else if(num_transmissions_>0 && p_buffer_ == 0) {setErrorCode(Error::INVALID_MESSAGE_BUFFER);}
200 // else if(num_transmissions_==0) {setErrorCode(Error::ZERO_SIZE_MESSAGE);} ///< I now use 0 size messages to ping a slave
201 
202 
203  else // TRANSMISSION HAS NO ERRORS
204  {
205  num_transmitted_ = 0;
206  send_register_address_pending_ = send_register_address_;
207 
208 
209  sendStart();
210  awaitTransmissionEnd();
211 
212 
213  // CHECK FOR ERRORS OCCURRED DURING TRANSMISSION
214  if(this->hasError()) {} // Avoid overwriting error code
215  else if(pendingTransmissions() > 0) {setErrorCode(Error::TRANSMISSION_PREMATURELY_ENDED);}
216  }
217 
218 
219  return not hasError(); // Returns true if no error occurred
220 }
